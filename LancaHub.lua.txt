local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoidRootPart = character:WaitForChild("HumanoidRootPart")

-- ===================== CONFIGURAÇÕES =====================

local DISTANCIA_ALVO = 5                          -- studs que a lança deve percorrer horizontalmente
local GRAVIDADE = workspace.Gravity               -- usa a gravidade atual do workspace (padrão: 196.2)
local ANGULO_LANCAMENTO = math.rad(45)            -- ângulo de 45° para alcance máximo
local COOLDOWN = 2                                -- segundos entre lançamentos

-- Calcula a velocidade inicial necessária para alcançar DISTANCIA_ALVO studs
-- Fórmula balística: d = v² * sin(2θ) / g  →  v = sqrt(d * g / sin(2θ))
local velocidadeInicial = math.sqrt(DISTANCIA_ALVO * GRAVIDADE / math.sin(2 * ANGULO_LANCAMENTO))

-- ===================== FUNÇÃO: CRIAR LANÇA =====================

local function criarLanca()
    -- Garante referências atualizadas após respawn
    character = player.Character or player.CharacterAdded:Wait()
    humanoidRootPart = character:WaitForChild("HumanoidRootPart")

    local lanca = Instance.new("Part")
    lanca.Name = "LancaProjetil"
    lanca.Size = Vector3.new(0.35, 0.35, 3.2)      -- comprimento e espessura da lança
    lanca.Shape = Enum.PartType.Cylinder
    lanca.BrickColor = BrickColor.new("Bright orange")
    lanca.Material = Enum.Material.Metal
    lanca.Reflectance = 0.3
    lanca.CastShadow = true
    lanca.Anchored = false
    lanca.CanCollide = true
    lanca.CollisionGroup = "Default"

    -- Posiciona na frente do personagem, levemente acima
    lanca.CFrame = humanoidRootPart.CFrame
        * CFrame.new(0, 1.2, -2.5)
        * CFrame.Angles(0, math.pi / 2, math.pi / 2)

    lanca.Parent = workspace

    -- Ponta da lança (cone visual)
    local ponta = Instance.new("SpecialMesh")
    ponta.MeshType = Enum.MeshType.FileMesh
    ponta.MeshId = "rbxasset://fonts/rightarm.mesh" -- placeholder; pode substituir por mesh real
    ponta.Parent = lanca

    return lanca
end

-- ===================== FUNÇÃO: LANÇAR LANÇA =====================

local podelancar = true
local heartbeatConexao = nil

local function lancarLanca()
    if not podelancar then return end
    podelancar = false

    -- Atualiza UI para cooldown
    botao.Text = "⏳  Aguarde..."
    botao.BackgroundColor3 = Color3.fromRGB(90, 45, 0)
    statusLabel.Text = "🕐 Cooldown ativo..."
    statusLabel.TextColor3 = Color3.fromRGB(255, 180, 50)

    -- Cria e posiciona a lança
    local lanca = criarLanca()

    -- Calcula vetor de velocidade com ângulo balístico
    local lookDir = humanoidRootPart.CFrame.LookVector
    local velHoriz = velocidadeInicial * math.cos(ANGULO_LANCAMENTO)
    local velVert  = velocidadeInicial * math.sin(ANGULO_LANCAMENTO)

    local velocidadeVetor = Vector3.new(
        lookDir.X * velHoriz,
        velVert,
        lookDir.Z * velHoriz
    )

    -- Aplica impulso via BodyVelocity
    local bodyVel = Instance.new("BodyVelocity")
    bodyVel.MaxForce = Vector3.new(1e6, 1e6, 1e6)
    bodyVel.Velocity = velocidadeVetor
    bodyVel.Parent = lanca

    -- Remove o BodyVelocity após um breve instante (física natural toma conta)
    task.delay(0.08, function()
        if bodyVel and bodyVel.Parent then
            bodyVel:Destroy()
        end
    end)

    -- Desconecta heartbeat anterior se existir
    if heartbeatConexao then
        heartbeatConexao:Disconnect()
        heartbeatConexao = nil
    end

    -- Aponta a lança na direção do vetor de velocidade em tempo real
    heartbeatConexao = RunService.Heartbeat:Connect(function()
        if lanca and lanca.Parent then
            local vel = lanca.AssemblyLinearVelocity
            if vel.Magnitude > 0.5 then
                local ok, _ = pcall(function()
                    lanca.CFrame = CFrame.lookAt(lanca.Position, lanca.Position + vel)
                        * CFrame.Angles(0, math.pi / 2, math.pi / 2)
                end)
                if not ok then
                    heartbeatConexao:Disconnect()
                    heartbeatConexao = nil
                end
            end
        else
            -- Lança destruída ou removida
            if heartbeatConexao then
                heartbeatConexao:Disconnect()
                heartbeatConexao = nil
            end
        end
    end)

    -- Destrói a lança automaticamente após 6 segundos
    task.delay(6, function()
        if heartbeatConexao then
            heartbeatConexao:Disconnect()
            heartbeatConexao = nil
        end
        if lanca and lanca.Parent then
            lanca:Destroy()
        end
    end)

    -- Cooldown: conta regressiva no status
    local tempoRestante = COOLDOWN
    task.spawn(function()
        while tempoRestante > 0 do
            statusLabel.Text = "🕐 Cooldown: " .. tempoRestante .. "s"
            task.wait(1)
            tempoRestante -= 1
        end
    end)

    -- Reativa o botão após o cooldown
    task.delay(COOLDOWN, function()
        podelancar = true
        botao.Text = "⚡  LANÇAR"
        botao.BackgroundColor3 = Color3.fromRGB(255, 85, 0)
        statusLabel.Text = "✅ Pronto para lançar!"
        statusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
    end)
end

-- ===================== EVENTO DO BOTÃO =====================

botao.MouseButton1Click:Connect(function()
    lancarLanca()
end)

-- ===================== RESPAWN: ATUALIZA REFERÊNCIAS =====================

    character = newChar
    humanoidRootPart = newChar:WaitForChild("HumanoidRootPart")
    -- Reseta o estado ao respawnar
    podelancar = true
    botao.Text = "⚡  LANÇAR"
